local function ConvertCharacterToR6(char)
    if not char then return nil end
    local hum = char:FindFirstChildOfClass("Humanoid")
    if not hum then return char end
    if hum.RigType == Enum.HumanoidRigType.R6 and char:FindFirstChild("Torso") then
        return char
    end

    local plr = Players:GetPlayerFromCharacter(char)
    local r6Model = nil

    pcall(function()
        local desc = nil
        if plr then
            pcall(function() desc = Players:GetHumanoidDescriptionFromUserId(plr.UserId) end)
        end
        if not desc then
            pcall(function() desc = hum:GetAppliedDescription() end)
        end
        if desc then
            r6Model = Players:CreateHumanoidModelFromDescription(desc, Enum.HumanoidRigType.R6)
        end
        if not r6Model and plr then
            r6Model = Players:CreateHumanoidModelFromUserId(plr.UserId, Enum.HumanoidRigType.R6)
        end
    end)

    if r6Model then
        r6Model.Name = char.Name
        local hrp = char:FindFirstChild("HumanoidRootPart")
        local newHrp = r6Model:FindFirstChild("HumanoidRootPart")
        if hrp and newHrp then
            newHrp.CFrame = hrp.CFrame
        end
        r6Model.Parent = char.Parent or Workspace
        if plr then
            plr.Character = r6Model
        end
        char:Destroy()
        return r6Model
    end
    return char
end

local function ForceUnanchor(character)
    if not character then return end
    for _, part in ipairs(character:GetDescendants()) do 
        if part:IsA("BasePart") then 
            part.Anchored = false 
            part.AssemblyLinearVelocity = Vector3.zero
            part.AssemblyAngularVelocity = Vector3.zero
        end 
    end
    local hum = character:FindFirstChildOfClass("Humanoid")
    if hum then
        hum.PlatformStand = false
        hum:SetStateEnabled(Enum.HumanoidStateType.Climbing, true)
        hum.AutoRotate = true
        hum.WalkSpeed = 16
        hum:ChangeState(Enum.HumanoidStateType.GettingUp)
    end
end

local function RemoveMorph(character)
    if not character then return end
    if character:FindFirstChild("Morph") then character.Morph:Destroy() end
    if character:FindFirstChild("ClothFolder") then 
        for _, v in character.ClothFolder:GetChildren() do v.Parent = character end
        character.ClothFolder:Destroy() 
    end
    if character:FindFirstChild("Torso") then character.Torso.Transparency = 0 end
    if character:FindFirstChild("Left Leg") then character["Left Leg"].Transparency = 0 end
    if character:FindFirstChild("Right Leg") then character["Right Leg"].Transparency = 0 end
    ForceUnanchor(character)
end

local function CleanUpCharacter(character, keepMorph)
    if not character then return end
    if not keepMorph then RemoveMorph(character) else ForceUnanchor(character) end
    for _, v in ipairs(character:GetDescendants()) do
        if v:IsA("NoCollisionConstraint") and v.Name == "ROCONS_NoCollide" then v:Destroy() end
    end
    local animateScript = character:FindFirstChild("Animate")
    if animateScript then animateScript.Disabled = false end
    for _, desc in ipairs(character:GetDescendants()) do
        if desc:IsA("Motor6D") then desc.Transform = CFrame.new() end
    end
end

local function ApplyMorph(character, morph, gender, keepClothing)
    if not character or not character:FindFirstChild("Torso") then return end
    local torso = character.Torso
    RemoveMorph(character)

    local morphClone = morph:Clone()
    morphClone.Name = "Morph"
    local clothFolder = character:FindFirstChild("ClothFolder") or Instance.new("Folder")
    clothFolder.Name = "ClothFolder"; clothFolder.Parent = character
    
    if not keepClothing then 
        for _, v in character:GetChildren() do 
            if v:IsA("Clothing") or v:IsA("ShirtGraphic") then 
                v.Parent = clothFolder 
            end 
        end 
    end

    for _, v in morphClone:GetDescendants() do
        if v:GetAttribute("Skin") then v.Color = torso.Color end
        if v.Parent == morphClone and (v:IsA("Weld") or v:IsA("Motor6D")) then
            v.Part0 = torso
            if gender == "FB" and v.Part1 and (v.Part1.Name == "Torso" or v.Part1.Name == "UpperTorso") then v.C0 = v.C0 * CFrame.Angles(0, math.rad(180), 0) end
        end
    end

    if gender == "FB" then
        for _, joint in ipairs(morphClone:GetDescendants()) do
            if (joint:IsA("Weld") or joint:IsA("Motor6D")) then
                if joint.Name:match("Hip") or joint.Name:match("Leg") or (joint.Part1 and joint.Part1.Name:match("Leg")) then joint.C0 = joint.C0 * CFrame.Angles(math.rad(180), 0, 0) end
            end
        end
    end

    for _, part in ipairs(morphClone:GetDescendants()) do
        if part:IsA("BasePart") then
            part.CanCollide = false; part.Massless = true
            local isWelded = false
            for _, weld in ipairs(morphClone:GetDescendants()) do if (weld:IsA("WeldConstraint") or weld:IsA("Weld") or weld:IsA("Motor6D")) and (weld.Part0 == part or weld.Part1 == part) then isWelded = true break end end
            if not isWelded then local wc = Instance.new("WeldConstraint"); wc.Part0 = torso; wc.Part1 = part; wc.Parent = part end
        end
    end

    if not keepClothing then 
        if gender == "M" then
            torso.Transparency = 0
        else
            torso.Transparency = 1 
        end
    else 
        if morphClone:FindFirstChild("Torso") then morphClone.Torso:Destroy() end 
    end

    for _, child in morphClone:GetChildren() do
        if child:FindFirstChild("NewLeftLeg") and character:FindFirstChild("Left Leg") then
            local leftLeg = child:FindFirstChild("NewLeftLeg"); leftLeg.Anchored = false; leftLeg.CFrame = character["Left Leg"].CFrame
            if leftLeg:FindFirstChild("WeldConstraint") then leftLeg.WeldConstraint.Part0 = character["Left Leg"] end
            character["Left Leg"].Transparency = 1
        elseif child:FindFirstChild("NewRightLeg") and character:FindFirstChild("Right Leg") then
            local rightLeg = child:FindFirstChild("NewRightLeg"); rightLeg.Anchored = false; rightLeg.CFrame = character["Right Leg"].CFrame
            if rightLeg:FindFirstChild("WeldConstraint") then rightLeg.WeldConstraint.Part0 = character["Right Leg"] end
            character["Right Leg"].Transparency = 1
        end
    end

    for _, part in ipairs(morphClone:GetDescendants()) do
        if part:IsA("BasePart") then
            part.Anchored = false
            if part.Name:match("Left") and part.Name:match("Leg") and character:FindFirstChild("Left Leg") then part.CFrame = character["Left Leg"].CFrame
            elseif part.Name:match("Right") and part.Name:match("Leg") and character:FindFirstChild("Right Leg") then part.CFrame = character["Right Leg"].CFrame
            else part.CFrame = torso.CFrame end
        end
    end
    morphClone.Parent = character
end

local function PlayKeyframeSequence(character, sequence, defaultSpeed)
    local animController = { Speed = defaultSpeed or 1.0, Stop = function() end }
    if not character then return animController end

    local animateScript = character:FindFirstChild("Animate")
    if animateScript then animateScript.Disabled = true end
    
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    if humanoid then
        humanoid.PlatformStand = true
        local animator = humanoid:FindFirstChildOfClass("Animator")
        if animator then for _, track in ipairs(animator:GetPlayingAnimationTracks()) do track:Stop() end end
    end

    local motors = {}
    for _, desc in ipairs(character:GetDescendants()) do 
        if desc:IsA("Motor6D") and desc.Part1 then motors[desc.Part1.Name] = desc end 
    end

    local validKeyframes = {}
    for _, child in ipairs(sequence:GetChildren()) do 
        if child:IsA("Keyframe") then table.insert(validKeyframes, child) end 
    end
    table.sort(validKeyframes, function(a, b) return a.Time < b.Time end)

    if #validKeyframes == 0 then return animController end

    local duration = validKeyframes[#validKeyframes].Time
    
    local keyframeData = {}
    for i, kf in ipairs(validKeyframes) do
        local poses = {}
        local function extractPoses(pose)
            local styleEnum = Enum.EasingStyle.Linear
            local dirEnum = Enum.EasingDirection.In
            pcall(function()
                styleEnum = Enum.EasingStyle[pose.EasingStyle.Name]
                dirEnum = Enum.EasingDirection[pose.EasingDirection.Name]
            end)
            
            poses[pose.Name] = {
                CFrame = pose.CFrame,
                Style = styleEnum,
                Dir = dirEnum
            }
            
            for _, subPose in ipairs(pose:GetSubPoses()) do 
                extractPoses(subPose) 
            end
        end
        for _, p in ipairs(kf:GetPoses()) do 
            extractPoses(p) 
        end
        
        keyframeData[i] = {
            Time = kf.Time,
            Poses = poses
        }
    end

    local isPlaying = true
    local elapsed = 0
    local lastTick = tick()
    local connection

    animController.Stop = function()
        isPlaying = false
        if connection then connection:Disconnect() end
        if humanoid then humanoid.PlatformStand = false end
        for _, motor in pairs(motors) do 
            if motor and motor.Parent then motor.Transform = CFrame.new() end 
        end
        if animateScript and animateScript.Parent then animateScript.Disabled = false end
    end

    connection = RunService.Stepped:Connect(function()
        if not isPlaying or not character or not character.Parent then animController.Stop(); return end
        local now = tick()
        local dt = now - lastTick
        lastTick = now
        
        elapsed = elapsed + (dt * animController.Speed)
        
        if elapsed > duration then 
            if sequence.Loop then 
                elapsed = elapsed % duration 
            else 
                animController.Stop(); return 
            end 
        end

        local kf1, kf2 = keyframeData[1], keyframeData[#keyframeData]
        for i = 1, #keyframeData - 1 do
            if elapsed >= keyframeData[i].Time and elapsed <= keyframeData[i+1].Time then 
                kf1, kf2 = keyframeData[i], keyframeData[i+1] 
                break 
            end
        end

        local alpha = 0
        if kf1 ~= kf2 and kf2.Time > kf1.Time then 
            alpha = (elapsed - kf1.Time) / (kf2.Time - kf1.Time) 
        end
        alpha = math.clamp(alpha, 0, 1)

        for motorName, motor in pairs(motors) do
            local pose1 = kf1.Poses[motorName]
            local pose2 = kf2.Poses[motorName]
            
            if pose1 then
                local cf = pose1.CFrame
                if pose2 then
                    local currentAlpha = alpha
                    if pose2.Style ~= Enum.EasingStyle.Linear then
                        currentAlpha = TweenService:GetValue(alpha, pose2.Style, pose2.Dir)
                    end
                    cf = pose1.CFrame:Lerp(pose2.CFrame, currentAlpha)
                end
                motor.Transform = cf
            end
        end
    end)
    return animController
end

local function DisableCollision(char1, char2)
    if not char1 or not char2 then return end
    for _, p1 in ipairs(char1:GetDescendants()) do
        if p1:IsA("BasePart") then
            for _, p2 in ipairs(char2:GetDescendants()) do
                if p2:IsA("BasePart") then
                    local ncc = Instance.new("NoCollisionConstraint")
                    ncc.Name = "ROCONS_NoCollide"; ncc.Part0 = p1; ncc.Part1 = p2; ncc.Parent = p1
                end
            end
        end
    end
end

local spawningBots = {}

local function CreateBotDummy(targetCharacter, botName, botUser, morphName, botRole)

    spawningBots[botName] = true

    for _, obj in ipairs(Workspace:GetChildren()) do

    end

    if not targetCharacter then 
        spawningBots[botName] = nil

    end

    targetCharacter = ConvertCharacterToR6(targetCharacter)
    
    local dummy
    if botUser and botUser ~= "" then
        local success, uid = pcall(function() return Players:GetUserIdFromNameAsync(botUser) end)
        if success and uid then
            local descSuccess, desc = pcall(function() return Players:GetHumanoidDescriptionFromUserId(uid) end)
            if descSuccess and desc then
                local modelSuccess, model = pcall(function() return Players:CreateHumanoidModelFromDescription(desc, Enum.HumanoidRigType.R6) end)
                if modelSuccess and model then
                    dummy = model
                end
            end
            if not dummy then
                local success2, model = pcall(function() return Players:CreateHumanoidModelFromUserId(uid, Enum.HumanoidRigType.R6) end)
                if success2 and model then
                    dummy = model
                end
            end
        end
    end
    
    if not dummy then
        targetCharacter.Archivable = true
        dummy = targetCharacter:Clone()
    end

local playerBubbleStacks = {}

local function CreateChatBubble(player, messageText)
    if not player or not player.Character then return end
    local head = player.Character:FindFirstChild("Head")
    if not head then return end

    if not playerBubbleStacks[player.Name] then playerBubbleStacks[player.Name] = {} end
    local stack = playerBubbleStacks[player.Name]

    if #stack >= 3 then
        local oldest = table.remove(stack, 1)
        if oldest and oldest.Gui then oldest.Gui:Destroy() end
    end

    local fontSize = 14
    local font = Enum.Font.GothamMedium
    local maxTextWidth = 140

    local textBounds = TextService:GetTextSize(messageText, fontSize, font, Vector2.new(maxTextWidth, 2000))
    local finalWidth = math.max(60, textBounds.X + 24)
    local finalHeight = textBounds.Y + 16

    local STUDS_PER_PIXEL = 0.04 

    local bubbleGui = Instance.new("BillboardGui")
    bubbleGui.Name = "ROCONS_ChatBubble"
    bubbleGui.Adornee = head
    bubbleGui.Size = UDim2.new(finalWidth * STUDS_PER_PIXEL, 0, finalHeight * STUDS_PER_PIXEL, 0) 
    bubbleGui.AlwaysOnTop = true
    bubbleGui.Parent = head

    local bubbleFrame = Instance.new("Frame", bubbleGui)
    bubbleFrame.Size = UDim2.new(0, 0, 0, 0)
    bubbleFrame.Position = UDim2.new(0.5, 0, 0.5, 0)
    bubbleFrame.AnchorPoint = Vector2.new(0.5, 0.5)
    bubbleFrame.BackgroundColor3 = Color3.fromRGB(30, 20, 40)
    bubbleFrame.BackgroundTransparency = 0.15 
    bubbleFrame.ClipsDescendants = true

    local bubbleCorner = Instance.new("UICorner", bubbleFrame)
    bubbleCorner.CornerRadius = UDim.new(0.25, 0)

    local txtLabel = Instance.new("TextLabel", bubbleFrame)
    txtLabel.Size = UDim2.new(textBounds.X / finalWidth, 0, textBounds.Y / finalHeight, 0)
    txtLabel.Position = UDim2.new(0.5, 0, 0.5, 0)
    txtLabel.AnchorPoint = Vector2.new(0.5, 0.5)
    txtLabel.BackgroundTransparency = 1
    txtLabel.Font = font
    txtLabel.TextScaled = true
    txtLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    txtLabel.TextTransparency = 1
    txtLabel.TextWrapped = true
    txtLabel.Text = messageText

    TweenService:Create(bubbleFrame, TweenInfo.new(0.35, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
        Size = UDim2.new(1, 0, 1, 0)
    }):Play()
    TweenService:Create(txtLabel, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
        TextTransparency = 0
    }):Play()

    local bubbleEntry = { Gui = bubbleGui, CreatedAt = tick(), Height = finalHeight * STUDS_PER_PIXEL }
    table.insert(stack, bubbleEntry)

    local currentYOffset = 3.8
    for i = #stack, 1, -1 do
        local item = stack[i]
        item.Gui.StudsOffset = Vector3.new(0, currentYOffset, 0)
        local studHeight = item.Height + 0.3
        currentYOffset = currentYOffset + studHeight
    end

    task.delay(7, function()
        if bubbleGui and bubbleGui.Parent then
            local fade1 = TweenService:Create(bubbleFrame, TweenInfo.new(0.5), {BackgroundTransparency = 1, Size = UDim2.new(0, 0, 0, 0)})
            local fade2 = TweenService:Create(txtLabel, TweenInfo.new(0.5), {TextTransparency = 1})
            fade1:Play(); fade2:Play()
            
            fade1.Completed:Connect(function()
                for i, entry in ipairs(stack) do
                    if entry.Gui == bubbleGui then
                        table.remove(stack, i)
                        break
                    end
                end
                bubbleGui:Destroy()
            end)
        end
    end)
end

local function SwitchChatSection(secName)
    activeSection = secName
    local channel = chatChannels[secName]
    if channel then
        channel.Unread = false
        if channel.Button then
            channel.Button.TextColor3 = Theme.TextDarkened
        end
    end

    for _, child in ipairs(ChatLogScroll:GetChildren()) do
        if child:IsA("Frame") then child:Destroy() end
    end

    if channel and channel.Messages then
        for _, msg in ipairs(channel.Messages) do
            local msgFrame = Instance.new("Frame", ChatLogScroll)
            msgFrame.Size = UDim2.new(1, -8, 0, 20)
            msgFrame.AutomaticSize = Enum.AutomaticSize.Y
            msgFrame.BackgroundTransparency = 1
            msgFrame.ZIndex = 17

            local msgBtn = Instance.new("TextButton", msgFrame)
            msgBtn.Size = UDim2.new(1, 0, 0, 20)
            msgBtn.AutomaticSize = Enum.AutomaticSize.Y
            msgBtn.BackgroundTransparency = 1
            msgBtn.Font = Enum.Font.Gotham
            msgBtn.TextSize = 11
            msgBtn.TextXAlignment = Enum.TextXAlignment.Left
            msgBtn.RichText = true
            msgBtn.TextWrapped = true
            msgBtn.ZIndex = 18

            local senderDisplayName = msg.DisplayName or msg.Sender or "Unknown"
            local rankStr = GetPlayerRank(msg.UserId)
            local formattedText = ""
            
            if rankStr and RankData.Colors[rankStr] then
                local c = RankData.Colors[rankStr]
                local hex = string.format("#%02X%02X%02X", math.floor(c.R*255), math.floor(c.G*255), math.floor(c.B*255))
                formattedText = string.format('<font color="%s"><b>[%s] %s</b></font>: <font color="#FFFFFF">%s</font>', hex, rankStr, senderDisplayName, msg.Text)
            else
                formattedText = string.format('<font color="#FFFFFF"><b>%s</b></font>: <font color="#FFFFFF">%s</font>', senderDisplayName, msg.Text)
            end
            
            msgBtn.Text = formattedText

            msgBtn.MouseButton1Click:Connect(function()
                if msg.Sender ~= localPlayer.Name then
                    OpenPrivateChat(msg.Sender)
                end
            end)
        end
    end
    
    task.wait(0.02)
    ChatLogScroll.CanvasSize = UDim2.new(0, 0, 0, ChatLogLayout.AbsoluteContentSize.Y)
    ChatLogScroll.CanvasPosition = Vector2.new(0, ChatLogScroll.CanvasSize.Y.Offset)
end

local function SendChatMessage()
    local text = ChatInput.Text
    if text:match("^%s*$") then return end
    ChatInput.Text = ""

    local msgObj = {
        Sender = localPlayer.Name,
        DisplayName = localPlayer.DisplayName,
        UserId = localPlayer.UserId,
        Text = text,
        Time = tick()
    }

    CreateChatBubble(localPlayer, text)

    if activeSection == "General" then
        FirebaseSet(string.format("Chat/General/%s/%d", game.JobId, math.floor(tick()*1000)), msgObj)
    elseif activeSection == "Global" then
        FirebaseSet(string.format("Chat/Global/%d", math.floor(tick()*1000)), msgObj)
    elseif activeSection:sub(1, 5) == "[PM] " then
        local pmTarget = activeSection:sub(6)
        FirebaseSet(string.format("Chat/Private/%s/%s/%d", localPlayer.Name, pmTarget, math.floor(tick()*1000)), msgObj)
        FirebaseSet(string.format("Chat/Private/%s/%s/%d", pmTarget, localPlayer.Name, math.floor(tick()*1000)), msgObj)
    end
end

local processedMsgKeys = {}
local scriptStartTime = tick() 

task.spawn(function()
    while true do
        task.wait(0.3)
        
        local genData = FirebaseGet("Chat/General/" .. game.JobId)
        if genData then
            for key, msg in pairs(genData) do
                if not processedMsgKeys["Gen_" .. key] then
                    processedMsgKeys["Gen_" .. key] = true
                    
                    if msg.Time and msg.Time >= scriptStartTime then
                        table.insert(chatChannels["General"].Messages, msg)
                        
                        if msg.Sender ~= localPlayer.Name then
                            local senderPlr = Players:FindFirstChild(msg.Sender)
                            if senderPlr then CreateChatBubble(senderPlr, msg.Text) end
                        end

                        if activeSection == "General" then
                            SwitchChatSection("General")
                        else
                            chatChannels["General"].Unread = true
                            if chatChannels["General"].Button then chatChannels["General"].Button.TextColor3 = Theme.TextPrimary end
                        end
                    end
                end
            end
        end

        local globData = FirebaseGet("Chat/Global")
        if globData then
            for key, msg in pairs(globData) do
                if not processedMsgKeys["Glob_" .. key] then
                    processedMsgKeys["Glob_" .. key] = true
                    
                    if msg.Time and msg.Time >= scriptStartTime then
                        table.insert(chatChannels["Global"].Messages, msg)

                        if msg.Sender ~= localPlayer.Name then
                            local senderPlr = Players:FindFirstChild(msg.Sender)
                            if senderPlr then CreateChatBubble(senderPlr, msg.Text) end
                        end

                        if activeSection == "Global" then
                            SwitchChatSection("Global")
                        else
                            chatChannels["Global"].Unread = true
                            if chatChannels["Global"].Button then chatChannels["Global"].Button.TextColor3 = Theme.TextPrimary end
                        end
                    end
                end
            end
        end

        local pmData = FirebaseGet("Chat/Private/" .. localPlayer.Name)
        if pmData then
            for senderName, msgList in pairs(pmData) do
                if not leftPlayers[senderName] then
                    local pmSecName = "[PM] " .. senderName
                    if not chatChannels[pmSecName] then CreateSectionTab(pmSecName) end
                    
                    for key, msg in pairs(msgList) do
                        if not processedMsgKeys["PM_" .. key] then
                            processedMsgKeys["PM_" .. key] = true
                            
                            if msg.Time and msg.Time >= scriptStartTime then
                                table.insert(chatChannels[pmSecName].Messages, msg)

                                if msg.Sender ~= localPlayer.Name then
                                    local senderPlr = Players:FindFirstChild(msg.Sender)
                                    if senderPlr then CreateChatBubble(senderPlr, msg.Text) end
                                end

                                if activeSection == pmSecName then
                                    SwitchChatSection(pmSecName)
                                else
                                    chatChannels[pmSecName].Unread = true
                                    if chatChannels[pmSecName].Button then chatChannels[pmSecName].Button.TextColor3 = Theme.TextPrimary end
                                end
                            end
                        end
                    end
                end
            end
        end
    end

local function StopEmote()
    if soloEmoteTrack then soloEmoteTrack.Stop(); soloEmoteTrack = nil end
    if currentState == "Emoting" then
        if localPlayer.Character then CleanUpCharacter(localPlayer.Character, isMorphed) end
        currentState = "Idle"
        FirebasePatch("Players/" .. localPlayer.Name, { Status = "Idle", CurrentAnim = "" })
    end
end

    FirebasePatch("Players/" .. localPlayer.Name, { Morphed = true })
    local myMorph = morphsFolder:FindFirstChild(chosenGender)
    if myMorph and localPlayer.Character then ApplyMorph(localPlayer.Character, myMorph, chosenGender, false) end
    
    MorphToggleBtn.Text = "Synced"; MorphToggleBtn.BackgroundColor3 = Theme.Panel
    
    task.delay(0.25, function()
        if localPlayer.Character and currentState == "Synced" then
            local myHRP = localPlayer.Character:FindFirstChild("HumanoidRootPart")
            local hum = localPlayer.Character:FindFirstChildOfClass("Humanoid")
            if hum then hum:SetStateEnabled(Enum.HumanoidStateType.Climbing, false); hum.AutoRotate = false end
            if myHRP then myHRP.AssemblyLinearVelocity = Vector3.zero; myHRP.AssemblyAngularVelocity = Vector3.zero; myHRP.Anchored = true end
        end
    end)
end

local function BreakSyncLocally()
    if currentState == "Idle" and not soloEmoteTrack then return end
    if currentState == "Emoting" then StopEmote(); return end

    ignoredPartner = syncPartner
    lastBreakTime = tick()

    currentState = "Idle"; myRole = ""; currentlyPlayingAnim = ""
    if currentTrack then currentTrack.Stop(); currentTrack = nil end
    if partnerTrack then partnerTrack.Stop(); partnerTrack = nil end
    
    if localPlayer.Character then CleanUpCharacter(localPlayer.Character, false) end
    
    local targetPartner = syncPartner
    if targetPartner ~= "" and targetPartner ~= "[BOT]" then
        local pPlayer = Players:FindFirstChild(targetPartner)
        if pPlayer and pPlayer.Character then CleanUpCharacter(pPlayer.Character, false) end
    end

    isMorphed = false
    MorphToggleBtn.Text = "Morph: Off"; MorphToggleBtn.BackgroundColor3 = Theme.Card
    TransferBtn.Visible = false
    
    currentAnimSpeed = 1.0
    SpeedLabel.Text = "Speed: 1.0x"
    currentControl = localPlayer.Name

    syncPartner = ""
    incomingRequester = ""
    incomingReqGender = ""

    local wipeData = { Status = "Idle", SyncPartner = "", CurrentAnim = "", Control = localPlayer.Name, AnimSpeed = 1.0, Role = "", Morphed = false, Request = "", RequesterGender = "" }
    FirebasePatch("Players/" .. localPlayer.Name, wipeData)
    if targetPartner ~= "" and targetPartner ~= "[BOT]" then FirebasePatch("Players/" .. targetPartner, wipeData) end
end

local function PlaySyncedAnimationsLocally(animName)
    local myFolder = animFolder:FindFirstChild(myRole)
    local partnerRole = (myRole == "D") and "S" or "D"
    local partnerFolder = animFolder:FindFirstChild(partnerRole)
    
    if myFolder and partnerFolder then
        local mySuffix = (myRole == "S") and "F" or "M"
        local partnerSuffix = (partnerRole == "S") and "F" or "M"
        
        local myKfs = myFolder:FindFirstChild(animName .. mySuffix)
        local partnerKfs = partnerFolder:FindFirstChild(animName .. partnerSuffix)
        
        if currentTrack then currentTrack.Stop() end
        if partnerTrack then partnerTrack.Stop() end
        
        if myKfs and localPlayer.Character then currentTrack = PlayKeyframeSequence(localPlayer.Character, myKfs, currentAnimSpeed) end
        
        if syncPartner == "[BOT]" then
            local botChar = Workspace:FindFirstChild("ROCONS_BOT_" .. localPlayer.Name)
            if botChar and partnerKfs then 
                partnerTrack = PlayKeyframeSequence(botChar, partnerKfs, currentAnimSpeed) 
            end
        else
            local pPlayer = Players:FindFirstChild(syncPartner)
            if pPlayer and pPlayer.Character and partnerKfs then partnerTrack = PlayKeyframeSequence(pPlayer.Character, partnerKfs, currentAnimSpeed) end
        end
    end
end

AcceptBtn.MouseButton1Click:Connect(function()
    if incomingRequester == "" then return end
    InviteModal.Visible = false
    StopEmote() 
    
    local reqRole, accRole = GetRoles(incomingReqGender, chosenGender)
    local targetRequester = incomingRequester 
    local controlPlayer = targetRequester 
    
    currentControl = controlPlayer
    lastControlState = controlPlayer
    TransferBtn.Visible = false 
    
    TweenDarkness(true, function()
        local reqPlayer = Players:FindFirstChild(targetRequester)
        local myChar = localPlayer.Character
        if reqPlayer and reqPlayer.Character and myChar then
            local reqIsMorphed = false
            if reqPlayer.Character:FindFirstChild("Morph") then reqIsMorphed = true end
            
            CleanUpCharacter(myChar, isMorphed)
            CleanUpCharacter(reqPlayer.Character, reqIsMorphed)

            local reqHRP = reqPlayer.Character:FindFirstChild("HumanoidRootPart")
            local myHRP = myChar:FindFirstChild("HumanoidRootPart")
            
            if reqHRP and myHRP then
                DisableCollision(myChar, reqPlayer.Character)
                myHRP.AssemblyLinearVelocity = Vector3.zero
                myHRP.AssemblyAngularVelocity = Vector3.zero
                
                myHRP.Anchored = true
                reqHRP.Anchored = true
                
                local offsetZ = (accRole == "S") and -1 or 1
                local targetPos = (reqHRP.CFrame * CFrame.new(0, 0, offsetZ)).Position
                local lookVec = reqHRP.CFrame.LookVector
                
                local fixedY = reqHRP.Position.Y
                local fixedTargetPos = Vector3.new(targetPos.X, fixedY, targetPos.Z)
                local lookAtPos = Vector3.new(fixedTargetPos.X + lookVec.X, fixedY, fixedTargetPos.Z + lookVec.Z)
                
                myHRP.CFrame = CFrame.lookAt(fixedTargetPos, lookAtPos)
                task.wait(0.05)
            end
        end
        
        local patchLocal = { Status = "Synced", SyncPartner = targetRequester, Role = accRole, Control = controlPlayer, AnimSpeed = 1.0, Request = "", RequesterGender = "", Morphed = true }
        local patchPartner = { Status = "Synced", SyncPartner = localPlayer.Name, Role = reqRole, Control = controlPlayer, AnimSpeed = 1.0, Request = "", RequesterGender = "", Morphed = true }

        FirebasePatch("Players/" .. localPlayer.Name, patchLocal)
        FirebasePatch("Players/" .. targetRequester, patchPartner)
        
        StartSyncLocal(targetRequester, accRole)
        incomingRequester = ""
        TweenDarkness(false)
    end)
end)
