local function updateStatusDisplay()
    StatusLabel.Text = string.format("X: %.2f | Y: %.2f | Z: %.2f\nJiggle: %.1fx", offsetX, offsetY, baseZOffset + offsetZ, jiggleMultiplier)
end

local function saveConfig()
    local fileContent = string.format("X_Offset=%.2f\nY_Offset=%.2f\nZ_Offset=%.2f\nMode=%s\nJiggle_Multiplier=%.1f", offsetX, offsetY, offsetZ, currentMode, jiggleMultiplier)
    
    local configFolder = Workspace:FindFirstChild("FunnyConfig")
    if not configFolder then
        configFolder = Instance.new("Folder")
        configFolder.Name = "FunnyConfig"
        configFolder.Parent = Workspace
    end
    
    local txtObject = configFolder:FindFirstChild("funny position.txt")
    if not txtObject then
        txtObject = Instance.new("StringValue")
        txtObject.Name = "funny position.txt"
        txtObject.Parent = configFolder
    end
    txtObject.Value = fileContent
    
    if writefile then
        pcall(function()
            writefile("funny position.txt", fileContent)
        end)
    end
    
    SaveButton.Text = "Saved!"
    task.delay(1.5, function()
        SaveButton.Text = "Save Config"
    end)
end

local function loadConfig()
    local content = nil
    
    if readfile then
        local success, result = pcall(readfile, "funny position.txt")
        if success and result then content = result end
    end
    
    if not content then
        local configFolder = Workspace:FindFirstChild("FunnyConfig")
        local txtObject = configFolder and configFolder:FindFirstChild("funny position.txt")
        if txtObject then content = txtObject.Value end
    end
    
    if content then
        local savedX = content:match("X_Offset=([%d.-]+)")
        local savedY = content:match("Y_Offset=([%d.-]+)")
        local savedZ = content:match("Z_Offset=([%d.-]+)")
        local savedMode = content:match("Mode=([%a%s]+)")
        local savedJiggle = content:match("Jiggle_Multiplier=([%d.-]+)")
        
        if savedX then offsetX = tonumber(savedX) or offsetX end
        if savedY then offsetY = tonumber(savedY) or offsetY end
        if savedZ then offsetZ = tonumber(savedZ) or offsetZ end
        if savedJiggle then jiggleMultiplier = tonumber(savedJiggle) or jiggleMultiplier end
        
        if savedMode then
            currentMode = savedMode
            if currentMode == "With Collision" then
                baseZOffset = -1.37
                ModeButton.Text = "Mode: Collision"
            else
                baseZOffset = -1.0
                ModeButton.Text = "Mode: No Collision"
            end
        end
        updateStatusDisplay()
    end
end

loadConfig()

local function stopAttachmentOnly()
    if animationTrack and animationTrack.IsPlaying then
        animationTrack:Stop(fadeTime)
    end
    
    local myChar = LocalPlayer.Character
    local myHumanoid = myChar and myChar:FindFirstChild("Humanoid")
    if myHumanoid then
        Camera.CameraSubject = myHumanoid
        Camera.CameraType = Enum.CameraType.Custom
    end
    
    local myRootPart = myChar and myChar:FindFirstChild("HumanoidRootPart")
    if myRootPart then
        myRootPart.Velocity = Vector3.zero
        myRootPart.RotVelocity = Vector3.zero
    end
end

local function stopEverything()
    stopAttachmentOnly()
    if scanConnection then
        scanConnection:Disconnect()
        scanConnection = nil
    end
    targetPlayer = nil
end

local function setupCharacter(character)
    local humanoid = character:WaitForChild("Humanoid", 10)
    if humanoid then
        animationTrack = humanoid:LoadAnimation(animation)
        humanoid.Died:Connect(function()
            stopEverything()
            ScreenGui:Destroy()
        end)
    end
end

if LocalPlayer.Character then
    setupCharacter(LocalPlayer.Character)
end

local function startMainLoop()
    if scanConnection then scanConnection:Disconnect() end
    scanConnection = RunService.Heartbeat:Connect(function(dt)
        if not isEnabled then return end
        
        if targetPlayer and not targetPlayer.Parent then
            targetPlayer = nil
            stopAttachmentOnly()
        end
        
        local myCurrentChar = LocalPlayer.Character
        local myRoot = myCurrentChar and myCurrentChar:FindFirstChild("HumanoidRootPart")
        if not myRoot then return end
        
        if not targetPlayer then
            for _, plr in ipairs(Players:GetPlayers()) do
                if plr ~= LocalPlayer and plr.Character then
                    local root = plr.Character:FindFirstChild("HumanoidRootPart")
                    local enemyHumanoid = plr.Character:FindFirstChild("Humanoid")
                    
                    if root and enemyHumanoid and enemyHumanoid.Health > 0 and (root.Position - myRoot.Position).Magnitude <= 5 then
                        targetPlayer = plr
                        jiggleRot = Vector3.zero
                        jiggleVel = Vector3.zero
                        lastTargetRot = CFrame.identity
                        break
                    end
                end
            end
        end
        
        if targetPlayer then
            local targetChar = targetPlayer.Character
            local targetRootPart = targetChar and targetChar:FindFirstChild("HumanoidRootPart")
            
            local targetTorso = targetChar and (targetChar:FindFirstChild("UpperTorso") or targetChar:FindFirstChild("Torso") or targetRootPart)
            local enemyHumanoid = targetChar and targetChar:FindFirstChild("Humanoid")
            
            if targetRootPart and targetTorso and enemyHumanoid and enemyHumanoid.Health > 0 then
                if Camera.CameraSubject ~= enemyHumanoid then
                    Camera.CameraType = Enum.CameraType.Custom
                    Camera.CameraSubject = enemyHumanoid
                end
                
                local activeHumanoid = myCurrentChar:FindFirstChild("Humanoid")
                if activeHumanoid and (not animationTrack or not animationTrack.IsPlaying) then
                    animationTrack = activeHumanoid:LoadAnimation(animation)
                    if animationTrack then
                        animationTrack:Play(fadeTime)
                        animationTrack.TimePosition = 1
                        animationTrack:AdjustSpeed(0)
                    end
                end
                
                if lastTargetRot == CFrame.identity then
                    lastTargetRot = targetTorso.CFrame.Rotation
                end
                
                pcall(function()
                    sethiddenproperty(myRoot, "PhysicsRepRootPart", targetRootPart)
                end)
                
                local targetVel = targetRootPart.AssemblyLinearVelocity
                local localVel = targetTorso.CFrame:VectorToObjectSpace(targetVel)
                
                local currentTargetRot = targetTorso.CFrame.Rotation
                local rotDelta = lastTargetRot:ToObjectSpace(currentTargetRot)
                local rx, ry, rz = rotDelta:ToEulerAnglesYXZ()
                local yawVelocity = ry / math.max(dt, 0.001)
                lastTargetRot = currentTargetRot
                
                local swayForce = ((localVel.X * 3.5) + (yawVelocity * 12.0)) * jiggleMultiplier
                
                local pitchForce = swayForce
                local rollForce = 0
                
                local force = Vector3.new(pitchForce, 0, rollForce)
                
                local springForce = -springProperties.stiffness * jiggleRot - springProperties.damping * jiggleVel + force
                jiggleVel = jiggleVel + springForce * dt
                jiggleRot = jiggleRot + jiggleVel * dt
                
                local cap = springProperties.maxRotation
                jiggleRot = Vector3.new(
                    math.clamp(jiggleRot.X, -cap, cap),
                    0,
                    math.clamp(jiggleRot.Z, -cap, cap)
                )
                
                local baseOffsetRotation = CFrame.Angles(0, 0, math.rad(90))
                local physicsJiggleRotation = CFrame.Angles(jiggleRot.X, 0, 0)
                
                local staticCFrame = (CFrame.new(targetTorso.Position) 
                    * targetTorso.CFrame.Rotation 
                    * CFrame.new(offsetX, offsetY, baseZOffset + offsetZ)) 
                    * baseOffsetRotation
                
                local pivotOffset = CFrame.new(0, 0, 0.5)
                
                myRoot.CFrame = staticCFrame * pivotOffset * physicsJiggleRotation * pivotOffset:Inverse()
                
                myRoot.Velocity = targetRootPart.AssemblyLinearVelocity
                myRoot.RotVelocity = targetRootPart.AssemblyAngularVelocity
            else
                stopAttachmentOnly()
                lastTargetRot = CFrame.identity
            end
        end
    end)
end
